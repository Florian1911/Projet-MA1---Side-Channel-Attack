A) Ajoute includes (dans /* USER CODE BEGIN Includes */)
#include <string.h>
#include "aes.h"

(On suppose que tu auras ajouté aes.c dans Core/Src et aes.h dans Core/Inc)

B) Ajoute variables AES (dans /* USER CODE BEGIN PV */)
static const uint8_t key[16] = {
  0x2B,0x7E,0x15,0x16, 0x28,0xAE,0xD2,0xA6,
  0xAB,0xF7,0x15,0x88, 0x09,0xCF,0x4F,0x3C
};

static const uint8_t pt0[16] = {
  0x6B,0xC1,0xBE,0xE2, 0x2E,0x40,0x9F,0x96,
  0xE9,0x3D,0x7E,0x11, 0x73,0x93,0x17,0x2A
};

static struct AES_ctx aes_ctx;

C) Init AES (dans /* USER CODE BEGIN 2 */)
AES_init_ctx(&aes_ctx, key);

D) Boucle AES + trigger (dans /* USER CODE BEGIN 3 */)
uint8_t buf[16];
memcpy(buf, pt0, 16);

HAL_GPIO_WritePin(TRIG_GPIO_Port, TRIG_Pin, GPIO_PIN_SET);
AES_ECB_encrypt(&aes_ctx, buf);
HAL_GPIO_WritePin(TRIG_GPIO_Port, TRIG_Pin, GPIO_PIN_RESET);

HAL_Delay(10);
